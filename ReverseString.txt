#include <iostream>
#include <cstring>
using namespace std;

void Reverse(char* str, int st, int en)
{
    int start = st, end = en;
    while(start <= end)
    {
        char temp = str[start];
        str[start] = str[end];
        str[end] = temp;
        start++;
        end--;
    }
}
void ReverseString(char* str)
{
    char* temp = str;
    int start = 0;
    int end = 0;
    while(*str != '\0') 
    {
        while( (*str != ' ') && *str != '\0' )
        {
            end++;
            str++;
        }
        Reverse(temp, start, end-1);
        start = end+1;
        end++;
        str++;
    }
    cout << "Reversed String Is :" << temp << endl;
}
